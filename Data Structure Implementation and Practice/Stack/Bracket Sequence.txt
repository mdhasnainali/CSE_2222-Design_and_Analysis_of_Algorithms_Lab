#include<bits/stdc++.h>
using namespace std;


int is_balanced(char input[])
{
	char stack[100], last_char;
	int top = 0;

	int length = strlen(input);

	for (int i = 0; i < length; i++)
	{
		if (input[i] == '(')
		{
			stack[top] = '(';
			top++;
		}
		else if (input[i] == ')')
		{
			if (top == 0)
			{
				return 0;
			}
			top--;

			/*last_char = stack[top];
			if (last_char != '(')
			{
				return 0;
			}
			*/
		}
	}

	if (top == 0) return 1;
	else return 0;
}

int main()
{
	freopen("Test-1-input.txt", "r", stdin);
	freopen("Test-1-output.txt", "w", stdout);

	char input[100];

	scanf("%s", input);

	if (is_balanced(input))
	{
		cout << "Balanced" << endl;
	}
	else
	{
		cout << "Not Blanced" << endl;
	}

	return 0;
}
